달력의 시작 날짜를 "월요일" 이나 특정 요일부터 시작하도록 설정한 모습입니다.("일요일"부터 시작이 기본값입니다.)

dataField: "field1",
headerText: "월요일",
dataType : "date",
dateInputFormat : "yyyy/mm/dd", // 실제 데이터의 형식 지정
formatString : "yyyy년 mm월 dd일", // 실제 데이터 형식을 어떻게 표시할지 지정
editRenderer : {
	type : "CalendarRenderer",
	defaultFormat : "yyyy/mm/dd", // 달력 선택 시 데이터에 적용되는 날짜 형식
	firstDay : 1, // 요일 시작 날짜 지정 (일요일인 경우 0, 월요일인 경우 1, 화요일인 경우 2 등)
