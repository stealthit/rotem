푸터(Footer)에 N개 행의 푸터를 출력한 모습입니다.

다수 행의 푸터를 출력하고자 한다면 푸터 출력 행 개수 속성인 footerRowCount 를 지정한 후 다음처럼 푸터를 정의할 때 2중 배열 구조로 정의합니다.

엑셀(xlsx)로 저장
// 그리드 속성 정의 var auiGridProps = { showFooter : true, // 푸터 출력 행 개수 설정 footerRowCount : 3, // 기본 푸터 높이 footerHeight : 24, // 푸터 행를 각각 20, 20, 30 으로 적용시킴 // 만약 지정하지 않는 경우 기본 footerHeight 이 적용됨 footerHeights : [20, 20, 30], ... }; // 푸터 레이아웃 정의 var footerLayout = []; // 첫번째 푸터 행 정의 footerLayout[0] = [{ positionField : "p1203", dataField : "p1203", }, { .... .... }]; // 두번째 푸터 행 정의 footerLayout[1] = [{ positionField : "p1203", dataField : "p1203", }, { .... ....
