푸터(Footer)에 다수 행를 설정한 후 세로 병합은 푸터 레이아웃 설정 시 rowSpan 의 개수를 지정하십시오.

푸터 변경하기 엑셀(xlsx)로 저장
// 그리드 속성 정의 // 푸터 레이아웃 정의 var footerLayout = []; // 첫번째 푸터 행 정의 footerLayout[0] = [{ positionField : "p1201", dataField : "p1201", labelText : "개별 세로(3)", style : "my-footer-style", rowSpan : 3 // 셀 세로 3개 병합 }, { positionField : "p1202", dataField : "p1202", labelText : "개별 세로(2)", style : "my-footer-style", rowSpan : 2 // 셀 세로 2개 병합 .... .... }]; // 두번째 푸터 행 정의 footerLayout[1] = [{ ... }, { positionField : "p1207", dataField : "p1207", labelText : "2X2 블럭-중", style : "my-footer-style", colSpan : 2, rowSpan : 2 // 가로, 세로 혼합 병합 (블럭 병합) }, { .... ....
